import { getCollection } from 'astro:content';
import Team from '../components/about/Team.astro';

export const team = (await getCollection('team')).map(member => member.data);
export const communityLinks = (await getCollection('community')).map(link => link.data);
export const podcasts = (await getCollection('podcasts')).map(podcast => podcast.data);
export const interviews = (await getCollection('interviews'))
    .filter(entry => entry.id !== 'index')
    .sort((a, b) => {
        const dateA = new Date(a.data.created || 0);
        const dateB = new Date(b.data.created || 0);
        return dateB.getTime() - dateA.getTime();
    });

htmx gives you access to AJAX, CSS Transitions, WebSockets and Server Sent Events directly in HTML, using attributes, so you can build modern user interfaces with the simplicity and power of hypertext.
