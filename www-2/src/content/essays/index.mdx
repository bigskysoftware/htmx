export const fmtDate = (d) => {
    if (!d) return undefined;
    return new Date(d).toLocaleDateString('en-US', {year: 'numeric', month: 'short', day: 'numeric'});
};

export const allEssays = [
    ...localEssays.map(e => ({
        title: e.data.title,
        url: `/essays/${e.id}`,
        author: e.data.authors?.join(', '),
        date: fmtDate(e.data.created),
        tags: Array.isArray(e.data.tags) ? e.data.tags.join(' ') : '',
        external: false,
    })),
    ...externalEssays.map(e => ({
        title: e.title,
        url: e.url,
        author: e.author,
        date: fmtDate(e.date),
        tags: e.tags,
        external: true,
    })),
].sort((a, b) => {
    if (!a.date && !b.date) return 0;
    if (!a.date) return 1;
    if (!b.date) return 1;
    return new Date(b.date).getTime() - new Date(a.date).getTime();
});
