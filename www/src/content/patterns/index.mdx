import { getCollection } from 'astro:content';
import FinderLink from '../../components/FinderLink.astro';
import Card from '../../components/Card.astro';

export const allEntries = await getCollection('patterns');

export const categoryEntries = allEntries
    .filter(e => !e.id.includes('/') && e.id !== 'index')
    .sort((a, b) => a.id.localeCompare(b.id));

export const patternEntries = allEntries
    .filter(e => e.id.includes('/'))
    .sort((a, b) => a.id.localeCompare(b.id));

export const cleanSlug = (id) => id.split('/').map(s => s.replace(/^\d+-/, '')).join('/');

export const groupedPatterns = categoryEntries.map(cat => ({
    slug: cleanSlug(cat.id),
    label: cat.data.title,
    items: patternEntries
        .filter(e => e.id.startsWith(cat.id + '/'))
        .map(e => ({
            title: e.data.title,
            url: `/patterns/${cleanSlug(e.id)}`,
            description: e.data.description,
            icon: e.data.icon || '',
            soon: e.data.soon || false,
        })),
}));

<div class="not-prose mt-6">
    {groupedPatterns.map(group => (
        <div class="mt-12 first:mt-0">
            <h2 class="font-sans text-xs font-medium uppercase tracking-widest text-neutral-600 dark:text-neutral-400 mb-6">
                {group.label}
            </h2>
            <div>
                <Card variant="lg" class="p-1.5">
                    <div class="flex flex-col gap-1.5">
                        {group.items.map(item => (
                            <FinderLink href={item.soon ? undefined : item.url} data-disabled={item.soon ? '' : undefined} class={`!mx-0${item.soon ? ' cursor-not-allowed pointer-events-none opacity-50' : ''}`}>
                                {item.icon && <i class={`${item.icon} size-5 shrink-0 mr-3 -translate-y-px group-interact/link:text-white dark:group-interact/link:text-black transition`}></i>}
                                <div class="flex flex-col sm:flex-row sm:items-center gap-1 sm:gap-5 flex-1 min-w-0">
                                    <span class="text-sm font-chicago tracking-tight sm:min-w-[16ch]">{item.title}</span>
                                    {item.description && (
                                        <span class="text-sm text-neutral-600 dark:text-neutral-400 group-interact/link:text-white dark:group-interact/link:text-black transition">{item.description}</span>
                                    )}
                                </div>
                                {item.soon && (
                                    <span class="text-[9px] px-1.5 py-0.5 shrink-0 self-start sm:self-center rounded bg-yellow-300/5 text-yellow-700 dark:text-yellow-300 border border-yellow-600/50 dark:border-yellow-300 font-medium uppercase tracking-wide">Soon</span>
                                )}
                            </FinderLink>
                        ))}
                    </div>
                </Card>
            </div>
        </div>
    ))}
</div>
